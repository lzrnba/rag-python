## 一、系统概述

### 1.1 系统定位

本系统是一个基于检索增强生成（RAG）技术的智能问答系统，专门用于技术文档的对话式查询。系统采用 Spring Boot（Java）+ Python 混合架构，Python 端负责核心的 RAG 推理服务。

### 1.2 技术栈总览

| 层级 | 技术选型 | 说明 |
|:--|:--|:--|
| **LLM** | Qwen2.5-7B-Instruct | 阿里云千问，通过 vLLM 部署 |
| **Embedding** | BAAI/bge-small-zh-v1.5 | 中文优化，768 维 |
| **Reranker** | BAAI/bge-reranker-v2-m3 | Cross-Encoder 重排 |
| **向量库** | FAISS | 支持 GPU 加速 |
| **关键词检索** | BM25 (rank_bm25) | 精确匹配 |
| **Agent 框架** | LangChain + LangGraph | 多 Agent 协作状态图 |
| **Web 框架** | FastAPI | RESTful API 服务 |
| **推理引擎** | vLLM | PagedAttention 并发优化 |

### 3.2 多 Agent 状态图 (agents/graph.py)

```python
from langgraph.graph import StateGraph, END
from typing import Literal

def create_agent_graph() -> StateGraph:
    """
    创建四智能体协作状态图
    
    流程：
    retrieve_rerank → grade → [sufficient] → generate → END
                              ↓
                         [insufficient] → rewrite → retrieve_rerank (循环)
    """
    workflow = StateGraph(AgentState)
    
    # 添加节点
    workflow.add_node("retrieve_rerank", retrieve_and_rerank_node)
    workflow.add_node("grade", grade_evidence_node)
    workflow.add_node("rewrite", rewrite_query_node)
    workflow.add_node("generate", generate_cot_answer_node)
    
    # 设置入口
    workflow.set_entry_point("retrieve_rerank")
    
    # 添加边
    workflow.add_edge("retrieve_rerank", "grade")
    
    # 条件路由：核心创新点
    workflow.add_conditional_edges(
        "grade",
        route_based_on_sufficiency,
        {
            "sufficient": "generate",
            "insufficient": "rewrite"
        }
    )
    
    workflow.add_edge("rewrite", "retrieve_rerank")  # 形成闭环
    workflow.add_edge("generate", END)
    
    return workflow.compile()

def route_based_on_sufficiency(state: AgentState) -> Literal["sufficient", "insufficient"]:
    """
    基于证据充分性的路由逻辑
    """
    MAX_ITER = state.get("max_iterations", 3)
    
    if state["is_sufficient"]:
        return "sufficient"
    
    if state["iterations"] >= MAX_ITER:
        # 达到最大迭代次数，强制生成
        return "sufficient"
    
    return "insufficient"
```

### 3.3 智能体节点实现 (agents/nodes.py)

#### 3.3.1 检索与重排节点

```python
from retrieval.hybrid import HybridRetriever
from retrieval.reranker import RerankerService

def retrieve_and_rerank_node(state: AgentState) -> AgentState:
    """
    检索与重排节点
    
    执行步骤：
    1. 向量检索召回（FAISS + Embedding）
    2. BM25 关键词检索
    3. 加权融合两种结果
    4. Cross-Encoder 深度重排
    5. 返回 Top-5 文档
    """
    query = state["query"]
    
    # 初始化检索器
    hybrid_retriever = HybridRetriever(
        vector_weight=0.7,
        bm25_weight=0.3,
        k=10  # 初筛 10 篇
    )
    
    # 混合检索
    initial_docs = hybrid_retriever.search(query)
    
    # Cross-Encoder 重排
    reranker = RerankerService(model_name="BAAI/bge-reranker-v2-m3")
    reranked_docs = reranker.rerank(query, initial_docs, top_k=5)
    
    # 更新状态
    state["documents"] = reranked_docs
    state["document_scores"] = [doc.get("score", 0.0) for doc in reranked_docs]
    state["iterations"] += 1
    
    return state
```

#### 3.3.2 评估智能体节点

```python
from llm.prompts import GRADER_PROMPT
from llm.vllm_client import QwenClient
import json

def grade_evidence_node(state: AgentState) -> AgentState:
    """
    评估智能体节点
    
    功能：
    - 判别当前文档是否足以回答问题
    - 输出布尔值及缺失信息描述
    - 使用 Qwen2.5-7B-Instruct 进行语义评估
    """
    client = QwenClient(model="Qwen/Qwen2.5-7B-Instruct")
    
    # 构建上下文
    context = "\n\n".join([doc["content"] for doc in state["documents"]])
    
    # 构建 Prompt
    prompt = GRADER_PROMPT.format(
        query=state["original_query"],
        context=context
    )
    
    # 调用 LLM
    response = client.generate(
        prompt=prompt,
        temperature=0.0,  # 评估需要确定性
        max_tokens=512
    )
    
    # 解析 JSON 响应
    try:
        result = json.loads(response.strip())
        state["is_sufficient"] = result.get("is_sufficient", False)
        state["missing_info"] = result.get("missing_info", "")
    except json.JSONDecodeError:
        # 解析失败时保守处理
        state["is_sufficient"] = False
        state["missing_info"] = "无法解析评估结果"
    
    return state
```

#### 3.3.4 生成智能体节点

```python
from llm.prompts import GENERATOR_PROMPT, STRUCTURED_PROMPT
from utils.term_dict import TermDictionary

def generate_cot_answer_node(state: AgentState) -> AgentState:
    """
    生成智能体节点
    
    功能：
    - 基于确认的上下文生成回答
    - 使用思维链（CoT）显式输出推理步骤
    - 术语一致性校验
    """
    client = QwenClient(model="Qwen/Qwen2.5-7B-Instruct")
    
    # 构建上下文
    context = "\n\n".join([doc["content"] for doc in state["documents"]])
    
    # 构建结构信息
    structure_info = "\n".join([
        doc.get("metadata", {}).get("section_path", "")
        for doc in state["documents"]
    ])
    
    # 使用结构化 Prompt
    prompt = STRUCTURED_PROMPT.format(
        structure_info=structure_info,
        context=context,
        query=state["original_query"]
    )
    
    # 调用 LLM
    response = client.generate(
        prompt=prompt,
        temperature=0.7,  # 生成需要一定创造性
        max_tokens=1024
    )
    
    # 解析 CoT 和最终答案
    content = response
    thought_match = re.search(r'<thought>(.*?)</thought>', content, re.DOTALL)
    
    if thought_match:
        state["reasoning"] = thought_match.group(1).strip()
        state["answer"] = content.split('</thought>')[-1].strip()
    else:
        state["answer"] = content.strip()
        state["reasoning"] = None
    
    # 术语一致性校验
    term_dict = TermDictionary()
    state["answer"] = term_dict.normalize(state["answer"])
    
    # 提取引用来源
    state["sources"] = [
        doc.get("metadata", {}).get("section_path", "")
        for doc in state["documents"]
    ]
    
    return state
```

### 3.4 混合检索实现 (retrieval/hybrid.py)

```python
from typing import List, Dict, Tuple
import numpy as np
from rank_bm25 import BM25Okapi

class HybridRetriever:
    """
    混合检索器：融合向量检索与 BM25 关键词检索
    """
    
    def __init__(
        self,
        vector_weight: float = 0.7,
        bm25_weight: float = 0.3,
        k: int = 5
    ):
        self.vector_weight = vector_weight
        self.bm25_weight = bm25_weight
        self.k = k
        
        # 初始化向量检索
        self.vector_store = self._init_vector_store()
        
        # 初始化 BM25
        self.bm25_retriever = self._init_bm25()
    
    def _init_vector_store(self):
        """初始化 FAISS 向量存储"""
        from langchain_community.vectorstores import FAISS
        from langchain_community.embeddings import HuggingFaceEmbeddings
        
        embedding_model = HuggingFaceEmbeddings(
            model_name="BAAI/bge-small-zh-v1.5",
            model_kwargs={'device': 'cuda'},
            encode_kwargs={'normalize_embeddings': True}
        )
        
        # 加载已保存的向量库
        return FAISS.load_local(
            folder_path="data/vectorstore",
            embeddings=embedding_model,
            allow_dangerous_deserialization=True
        )
    
    def _init_bm25(self):
        """初始化 BM25 检索器"""
        # 从向量库获取所有文档
        docs = [doc.page_content for doc in self.vector_store.docstore.values()]
        tokenized_docs = [doc.split() for doc in docs]
        return BM25Okapi(tokenized_docs)
    
    def search(self, query: str) -> List[Dict]:
        """
        执行混合检索
        
        步骤：
        1. 向量检索召回
        2. BM25 关键词检索
        3. 分数归一化
        4. 加权融合
        5. 返回 Top-K
        """
        # 向量检索
        query_vector = self.vector_store.embedding_model.embed_query(query)
        vector_results = self.vector_store.similarity_search_with_score(
            query_vector, k=self.k * 2
        )
        
        # BM25 检索
        tokenized_query = query.split()
        bm25_scores = self.bm25_retriever.get_scores(tokenized_query)
        top_indices = bm25_scores.argsort()[-self.k * 2:][::-1]
        bm25_results = [(i, bm25_scores[i]) for i in top_indices]
        
        # 归一化分数
        vector_scores = self._normalize_scores([s for _, s in vector_results])
        bm25_scores_norm = self._normalize_scores([s for _, s in bm25_results])
        
        # 加权融合
        doc_scores = {}
        for (doc, _), v_score, (_, b_score) in zip(
            vector_results, vector_scores, bm25_results
        ):
            doc_id = id(doc)
            if doc_id not in doc_scores:
                doc_scores[doc_id] = 0
            doc_scores[doc_id] += (
                self.vector_weight * v_score + 
                self.bm25_weight * b_score
            )
        
        # 排序返回
        sorted_docs = sorted(
            doc_scores.items(), key=lambda x: x[1], reverse=True
        )[:self.k]
        
        return [
            {"content": doc.page_content, "score": score, "metadata": doc.metadata}
            for doc_id, score in sorted_docs
        ]
    
    def _normalize_scores(self, scores: List[float]) -> List[float]:
        """Min-Max 归一化"""
        if not scores:
            return []
        min_score = min(scores)
        max_score = max(scores)
        if max_score == min_score:
            return [0.5] * len(scores)
        return [(s - min_score) / (max_score - min_score) for s in scores]
```

### 3.5 语义重排实现 (retrieval/reranker.py)

```python
from typing import List, Dict
from langchain_community.cross_encoders import HuggingFaceCrossEncoder

class RerankerService:
    """
    Cross-Encoder 语义重排服务
    """
    
    def __init__(self, model_name: str = "BAAI/bge-reranker-v2-m3"):
        self.reranker = HuggingFaceCrossEncoder(model_name=model_name)
    
    def rerank(
        self,
        query: str,
        documents: List[Dict],
        top_k: int = 5
    ) -> List[Dict]:
        """
        对文档进行重排
        
        Args:
            query: 查询文本
            documents: 待重排文档列表
            top_k: 返回 Top-K 结果
        
        Returns:
            重排后的文档列表
        """
        if not documents:
            return []
        
        # 构建 (query, doc) 对
        pairs = [(query, doc["content"]) for doc in documents]
        
        # 预测相关性分数
        scores = self.reranker.predict(pairs)
        
        # 按分数排序
        scored_docs = list(zip(documents, scores))
        sorted_docs = sorted(scored_docs, key=lambda x: x[1], reverse=True)
        
        # 返回 Top-K
        return [
            {**doc, "rerank_score": float(score)}
            for doc, score in sorted_docs[:top_k]
        ]
```

### 3.6 结构化文档解析 (parsing/general.py)

```python
import re
from typing import List, Dict, Optional

class GeneralParser:
    """
    General 结构化文档解析器
    
    自动识别技术文档的章节层级、类方法结构、图表引用关系
    """
    
    def __init__(self):
        # 定义层级模式
        self.patterns = {
            'chapter': r'^第 [一二三四五六七八九十]+章\s+(.+)$',
            'section': r'^(\d+\.)\s+(.+)$',
            'subsection': r'^(\d+\.\d+)\s+(.+)$',
            'subsubsection': r'^(\d+\.\d+\.\d+)\s+(.+)$',
            'class': r'^class\s+(\w+)',
            'method': r'^def\s+(\w+)\s*\(',
            'api': r'^`?(\w+\.\w+\.?\w*)`?',
        }
        
        self.current_structure = {
            'chapter': '',
            'section': '',
            'subsection': '',
            'class': '',
            'method': ''
        }
    
    def parse(self, text: str, doc_name: str = "") -> List[Dict]:
        """
        解析文档，保留层级结构
        
        Returns:
            包含结构信息的文档片段列表
        """
        chunks = []
        current_chunk = []
        
        lines = text.split('\n')
        
        for line in lines:
            # 检测层级变化
            structure_change = self.detect_structure(line)
            
            if structure_change:
                # 保存当前 chunk
                if current_chunk:
                    chunk_data = self._create_chunk(
                        '\n'.join(current_chunk),
                        doc_name
                    )
                    chunks.append(chunk_data)
                    current_chunk = []
                
                # 更新结构
                self.current_structure.update(structure_change)
            
            current_chunk.append(line)
        
        # 保存最后一个 chunk
        if current_chunk:
            chunk_data = self._create_chunk(
                '\n'.join(current_chunk),
                doc_name
            )
            chunks.append(chunk_data)
        
        return chunks
    
    def detect_structure(self, line: str) -> Optional[Dict]:
        """检测行所属的层级"""
        for level, pattern in self.patterns.items():
            match = re.match(pattern, line)
            if match:
                if match.lastindex:
                    return {level: match.group(1)}
                else:
                    return {level: match.group(0)}
        return {}
    
    def _create_chunk(self, content: str, doc_name: str) -> Dict:
        """创建包含元数据的文档片段"""
        return {
            "content": content,
            "structure": self.current_structure.copy(),
            "section_path": self.build_section_path(),
            "metadata": {
                "doc_name": doc_name,
                "chapter": self.current_structure.get("chapter", ""),
                "section": self.current_structure.get("section", ""),
            }
        }
    
    def build_section_path(self) -> str:
        """构建章节路径字符串"""
        parts = []
        for key in ['chapter', 'section', 'subsection', 'class', 'method']:
            if self.current_structure.get(key):
                parts.append(self.current_structure[key])
        return ' → '.join(parts)
```

```python
def create_text_splitter() -> RecursiveCharacterTextSplitter:
```

### 3.8 术语词典与一致性校验 (utils/term_dict.py)

```python
from typing import Dict, List

class TermDictionary:
    """
    技术术语词典
    
    功能：
    - 术语标准化
    - 别名映射
    - 生成后校验
    """
    
    def __init__(self):
        # 标准术语及其别名
        self.terms: Dict[str, List[str]] = {
            "HttpServletRequest": ["request", "请求对象"],
            "UserController": ["用户控制器", "User 控制器"],
            "login": ["登录", "登陆"],
            "authentication": ["认证", "鉴权", "身份验证"],
            "authorization": ["授权", "权限验证"],
            "RestController": ["REST 控制器", "RESTful 控制器"],
            "Autowired": ["自动装配", "依赖注入"],
            "Service": ["服务层", "业务逻辑层"],
            "Repository": ["数据访问层", "DAO"],
            "MyBatis-Plus": ["MyBatisPlus", "MP"],
            "FastAPI": ["Fast API", "Fast-API"],
            "LangChain": ["Lang Chain", "Lang-Chain"],
            "FAISS": ["Faiss", "faiss"],
            "BM25": ["BM-25", "bm25"],
        }
        
        # 构建反向索引：别名 -> 标准术语
        self.alias_to_standard: Dict[str, str] = {}
        for standard, aliases in self.terms.items():
            for alias in aliases:
                self.alias_to_standard[alias] = standard
    
    def normalize(self, text: str) -> str:
        """
        将文本中的术语标准化
        
        策略：
        - 按长度降序排序，优先替换长术语（避免部分替换）
        - 保留原始术语的大小写形式
        """
        result = text
        
        # 按别名长度降序排序
        sorted_aliases = sorted(
            self.alias_to_standard.keys(),
            key=len,
            reverse=True
        )
        
        for alias in sorted_aliases:
            standard = self.alias_to_standard[alias]
            if alias != standard:
                result = result.replace(alias, standard)
        
        return result
    
    def get_standard_term(self, alias: str) -> str:
        """获取标准术语"""
        return self.alias_to_standard.get(alias, alias)
```

### 3.9 LLM 客户端 (llm/vllm_client.py)

```python
from typing import Optional, List
import requests

class QwenClient:
    """
    Qwen2.5-7B-Instruct 客户端
    
    通过 vLLM 部署，提供 OpenAI 兼容的 API 接口
    """
    
    def __init__(
        self,
        model: str = "Qwen/Qwen2.5-7B-Instruct",
        base_url: str = "http://localhost:8000/v1",
        api_key: str = "EMPTY"
    ):
        self.model = model
        self.base_url = base_url
        self.api_key = api_key
        self.session = requests.Session()
    
    def generate(
        self,
        prompt: str,
        temperature: float = 0.7,
        max_tokens: int = 1024,
        top_p: float = 0.95,
        stop: Optional[List[str]] = None
    ) -> str:
        """
        生成文本
        
        Args:
            prompt: 输入提示词
            temperature: 温度参数（0.0 确定性，1.0 创造性）
            max_tokens: 最大生成长度
            top_p: 核采样参数
            stop: 停止词列表
        
        Returns:
            生成的文本
        """
        payload = {
            "model": self.model,
            "messages": [
                {"role": "user", "content": prompt}
            ],
            "temperature": temperature,
            "max_tokens": max_tokens,
            "top_p": top_p,
            "stop": stop or []
        }
        
        headers = {
            "Content-Type": "application/json",
            "Authorization": f"Bearer {self.api_key}"
        }
        
        response = self.session.post(
            f"{self.base_url}/chat/completions",
            json=payload,
            headers=headers,
            timeout=60
        )
        
        response.raise_for_status()
        result = response.json()
        
        return result["choices"][0]["message"]["content"]
    
    def embed(self, texts: List[str]) -> List[List[float]]:
        """
        获取文本嵌入向量
        
        注意：实际项目中应使用专门的 Embedding 模型
        """
        payload = {
            "model": "BAAI/bge-small-zh-v1.5",
            "input": texts
        }
        
        headers = {
            "Content-Type": "application/json",
            "Authorization": f"Bearer {self.api_key}"
        }
        
        response = self.session.post(
            f"{self.base_url}/embeddings",
            json=payload,
            headers=headers,
            timeout=60
        )
        
        response.raise_for_status()
        result = response.json()
        
        return [item["embedding"] for item in result["data"]]
```

```python
from fastapi import APIRouter, HTTPException, Depends
from pydantic import BaseModel, Field
from typing import List, Optional

from agents.graph import create_agent_graph
from core.config import settings

router = APIRouter(prefix="/v1", tags=["chat"])

# 数据模型
class ChatRequest(BaseModel):
    query: str = Field(..., description="用户问题", min_length=1)
    user_id: str = Field(..., description="用户 ID")
    conversation_id: Optional[str] = Field(None, description="会话 ID")

class ChatResponse(BaseModel):
    answer: str = Field(..., description="回答内容")
    reasoning: Optional[str] = Field(None, description="推理过程")
    sources: List[str] = Field(default_factory=list, description="引用来源")
    conversation_id: str = Field(..., description="会话 ID")

# 全局 RAG 图实例
rag_graph = None

@router.on_event("startup")
async def startup_event():
    """应用启动时初始化 RAG 图"""
    global rag_graph
    rag_graph = create_agent_graph()

@router.post("/chat/completions", response_model=ChatResponse)
async def chat_completion(request: ChatRequest):
    """
    RAG 对话接口
    
    流程：
    1. 构建初始状态
    2. 调用 RAG 状态图
    3. 返回响应
    """
    try:
        # 构建初始状态
        initial_state = {
            "query": request.query,
            "original_query": request.query,
            "user_id": request.user_id,
            "conversation_id": request.conversation_id or generate_conversation_id(),
            "documents": [],
            "document_scores": [],
            "is_sufficient": False,
            "missing_info": "",
            "iterations": 0,
            "max_iterations": 3,
            "answer": "",
            "reasoning": None,
            "sources": [],
            "error": None
        }
        
        # 调用 RAG 图
        result = rag_graph.invoke(initial_state)
        
        # 检查错误
        if result.get("error"):
            raise HTTPException(status_code=500, detail=result["error"])
        
        # 构建响应
        return ChatResponse(
            answer=result["answer"],
            reasoning=result.get("reasoning"),
            sources=result.get("sources", []),
            conversation_id=result["conversation_id"]
        )
        
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

def generate_conversation_id() -> str:
    """生成会话 ID"""
    import uuid
    return str(uuid.uuid4())
```

## 九、关键设计决策

### 9.1 为什么选择 LangGraph？

- **动态路由**：根据证据充分性决定流程走向，而非硬编码固定流程
- **闭环迭代**：支持"检索 - 评估 - 重写"的循环，直到满足终止条件
- **可追溯性**：状态变更可记录，便于调试和审计

### 9.2 为什么采用混合检索？

| 检索方式 | 优势 | 劣势 |
|:--|:--|:--|
| 向量检索 | 语义理解，解决"词汇鸿沟" | 专有名词精确匹配不足 |
| BM25 | 精确匹配术语、API 名称 | 无法理解语义 |
| **混合检索** | **兼顾两者，F1 提升 21%** | 实现复杂度略高 |

### 9.3 为什么使用 Qwen2.5-7B？

- **中文优化**：阿里云千问对中文场景有更好支持
- **适中规模**：7B 参数在性能和资源间取得平衡
- **开源免费**：可本地部署，无 API 调用成本
- **指令遵循**：Instruct 版本对 Prompt 指令遵循度高
